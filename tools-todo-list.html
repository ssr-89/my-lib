"use strict";
document.addEventListener("DOMContentLoaded", function () {
	// Поиск HTML-элементов
  const heroForm = document.querySelector(".hero-form");
  const heroInput = heroForm.querySelector(".hero-form__input");
  const heroWrapperFormBtn = heroForm.querySelector(".hero-form-block");
  const heroFormBtn = heroWrapperFormBtn.querySelector(".hero-form-block__btn");

  // Присваивание кнопке атрибута 'disabled'
  heroFormBtn.disabled = true;
  heroInput.addEventListener("input", function () {
    heroFormBtn.disabled = false;
  });

  // Поиск HTML-элементов
  const todoList = document.querySelector(".todo-list");

  // Создание пустого массива, который будет заполняться по мере добавления задач
  const todoListArr = new Array();

  /*функция создания элементов при указании задачи*/
  function createTodoItem(name) {
  	// Создание ячейки с задачей
    let todoListItem = document.createElement("li");

    // Присваивание ячейке класса и указания значения
    todoListItem.classList.add("todo-list__item");
    todoListItem.textContent = name; // Значение, введённое в heroInput

    // Создание блока функции-кнопок и присваивание классов
    let buttonGroup = document.createElement("div");
    buttonGroup.classList.add("todo-list__block-btns", "todo-list-block-btns");

    // Создание функции-кнопок и присваивание классов с текстом
    let doneBtn = document.createElement("button");
    doneBtn.classList.add(
      "todo-list-block-btns__btn",
      "todo-list-block-btns-ok",
      "btn"
    );
    doneBtn.textContent = "Ok";
    let delBtn = document.createElement("button");
    delBtn.classList.add(
      "todo-list-block-btns__btn",
      "todo-list-block-btns-del",
      "btn"
    );
    delBtn.textContent = "Del";

    // Добавление ранее созданных элементов в другие элементы
    buttonGroup.append(doneBtn);
    buttonGroup.append(delBtn);
    todoListItem.append(buttonGroup);

    // Добавление обработчиков событий на функции-кнопки
    doneBtn.addEventListener("click", function () {
      todoListItem.classList.toggle("todo-list__item--green");

      // Цикл для указания состояния объект на основе синхронности id ячейки с id объекта todoListObj
      for (let itemList of todoListArr) {
        if (itemList.id == todoListObj.id) itemList.done = !itemList.done; // Если id объект массива равен id ячейки, то свойство объекта done меняет булево-значение
      }
    });


    delBtn.addEventListener("click", function () {
      if (confirm("Are you sure?")) {
        todoListItem.remove();

        // Цикл по удалению нужного объекта
        // 1) если i меньше длины todoListArr    3) если второе условие не выполнено, то i прибавляется единица и переходит к первому условию
        for (let i = 0; i < todoListArr.length; i++) {

        	// 2) если id элемента i равен id todoListObj
          if (todoListArr[i].id == todoListObj.id) {

          	// 3) то элемент (объект) i и 1 (кол-во элементов, которое следует, начиная с i) удаляется
          	todoListArr.splice(i, 1);

          }
        }
      }
    });

    // Присваивание id ячейке
    todoListItem.id = itemId(todoListArr);

    // Функция создания id
    function itemId(arr) {
      let max = 0;
      for (let item of arr) {
        if (item.id > max) max = item.id;
      }
      return max + 1;
    }

    // Создание объекта массива с добавлением свойств со значениями ячейки
    let todoListObj = {
      id: itemId(todoListArr),
      name: heroInput.value,
      done: false
    };
    todoListArr.push(todoListObj);

    console.log(todoListArr);
    // console.log(todoListObj);

    return {
      todoListItem,
      doneBtn,
      delBtn
    };
  }

  // Функция по объединению всего выше изложенного и запуска процесса добавления задач
  function createTodoApp() {
    heroForm.addEventListener("submit", function (e) {
      e.preventDefault(); // предотвращает перезагрузку страницы при отправке формы

      // игнорируем создание элемента, если пользователь ничего не ввёл в поле
      if (!heroInput.value) {
        return;
      }

      // Переменная со значением функции по созданию ячейки и указанным параметром.
      let todoItem = createTodoItem(heroInput.value);

      // Добавление ячейки со значением heroInput в список задач
      todoList.append(todoItem.todoListItem);

      // Добавление атрибута 'disabled' кнопке
      heroFormBtn.setAttribute("disabled", "disabled");

      // Удаление значения heroInput после добавления задачи
      heroInput.value = "";
    });
  }

  createTodoApp();
});
